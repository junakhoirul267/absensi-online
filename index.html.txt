    <script>
        // ========== GANTI URL INI DENGAN WEB APP URL ANDA ==========
        const WEBAPP_URL = "https://script.google.com/macros/s/AKfycbxEl110RrPxVR6txYjrI2T5c5vtfNYjZksOhc_fkcL6taYXTQNSNYnLuL_qmmm1toBIQw/exec"; 
        // ============================================================
        
        const btnKirim = document.getElementById('btnKirim');
        const hasilDiv = document.getElementById('hasil');

        async function kirimAbsen() {
            const nama = document.getElementById('nama').value.trim();
            const status = document.getElementById('status').value;
            const keterangan = document.getElementById('keterangan').value.trim();

            // Validasi
            if (!nama) {
                tampilkanHasil('❌ Nama harus diisi!', 'error');
                return;
            }

            // Nonaktifkan tombol
            btnKirim.disabled = true;
            btnKirim.textContent = '⏳ Mengirim...';
            tampilkanHasil('⏳ Mengirim data...', 'info');

            try {
                const response = await fetch(WEBAPP_URL, {
                    method: 'POST',
                    mode: 'no-cors', // Penting untuk Apps Script
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({
                        nama: nama,
                        status: status,
                        keterangan: keterangan
                    })
                });

                // Karena mode no-cors, response tidak bisa dibaca
                // Tapi data tetap terkirim
                tampilkanHasil('✅ Absensi berhasil dikirim!', 'success');
                
                // Kosongkan field
                document.getElementById('nama').value = '';
                document.getElementById('keterangan').value = '';
                
            } catch (error) {
                tampilkanHasil('❌ Gagal mengirim: ' + error.message, 'error');
            } finally {
                btnKirim.disabled = false;
                btnKirim.textContent = '📤 Kirim Absen';
            }
        }

        function tampilkanHasil(pesan, tipe) {
            hasilDiv.textContent = pesan;
            hasilDiv.className = tipe;
        }

        // Fungsi untuk melihat data (bisa diakses via spreadsheet langsung)
        function lihatData() {
            // Arahkan ke spreadsheet
            const spreadsheetUrl = prompt(
                'Masukkan URL spreadsheet Anda untuk melihat data:\n' +
                '(Buka spreadsheet > klik Bagikan > copy link)'
            );
            if (spreadsheetUrl) {
                window.open(spreadsheetUrl, '_blank');
            }
        }

        // Enter key support
        document.addEventListener('keydown', function(e) {
            if (e.key === 'Enter') {
                kirimAbsen();
            }
        });

        console.log('✅ Situs Absensi siap digunakan!');
        console.log('📌 Web App URL:', WEBAPP_URL);
    </script>
